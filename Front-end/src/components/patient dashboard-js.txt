            <Form.Group controlId="formBasicDate">

    AppointmentService.getAppointmentsByPatientId(id)
      .then((response) => {
        if (response.status === 200) {
          const currentDate = new Date();
  
          const fetchDoctorName = async (appointment) => {
            try {
              const doctorResponse = await DoctorService.getDoctorNameById(appointment.doctorId);
              if (doctorResponse.status === 200) {
                appointment.doctorName = doctorResponse.data.name;
              } else {
                console.error('Error fetching doctor name:', doctorResponse.data);
              }
            } catch (error) {
              console.error('Error fetching doctor name:', error);
            }
          };
  
          const futureAppointments = response.data;
          // Fetch the doctor's name for each appointment
          const appointmentsWithDoctorNames = futureAppointments.map(fetchDoctorName);
  
          // Filter out appointments that are not resolved yet
          const filteredAppointments = appointmentsWithDoctorNames.filter(
            (appointment) => new Date(appointment.date) >= currentDate
          );

          console.log(filteredAppointments);
  
          setAppointments(filteredAppointments);
        } else {
          console.error('Error fetching appointments:', response.data);
        }
      })
      .catch((error) => {
        console.error('Error fetching appointments:', error);
      });
  };
